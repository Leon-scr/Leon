-- Sonic Animation Hub | By leon

local TweenService = game:GetService("TweenService")

-- CONFIG
local KEY = "anhdaideptrai"
local SCRIPT_URL = "https://raw.githubusercontent.com/catwljz/Scripts/refs/heads/main/SonicbyReapvitalized.txt"
local ICON_ID = "rbxassetid://80226210743615"

-- GUI
local gui = Instance.new("ScreenGui")
gui.IgnoreGuiInset = true
gui.ResetOnSpawn = false
gui.Parent = game.CoreGui

local bg = Instance.new("Frame", gui)
bg.Size = UDim2.fromScale(1,1)
bg.BackgroundColor3 = Color3.fromRGB(12,12,12)

-- ICON TRÒN (ĐỨNG IM)
local icon = Instance.new("ImageLabel", bg)
icon.Size = UDim2.fromScale(0.15,0.15)
icon.Position = UDim2.fromScale(0.5,0.35)
icon.AnchorPoint = Vector2.new(0.5,0.5)
icon.BackgroundTransparency = 1
icon.Image = ICON_ID
icon.ScaleType = Enum.ScaleType.Crop

Instance.new("UICorner", icon).CornerRadius = UDim.new(1,0)

-- STATUS TEXT
local status = Instance.new("TextLabel", bg)
status.Size = UDim2.fromScale(1,0.06)
status.Position = UDim2.fromScale(0.5,0.5)
status.AnchorPoint = Vector2.new(0.5,0.5)
status.BackgroundTransparency = 1
status.Font = Enum.Font.Gotham
status.TextSize = 18
status.TextColor3 = Color3.fromRGB(220,220,220)
status.Text = "Enter access key"

-- INPUT KEY
local input = Instance.new("TextBox", bg)
input.Size = UDim2.fromScale(0.3,0.065)
input.Position = UDim2.fromScale(0.5,0.58)
input.AnchorPoint = Vector2.new(0.5,0.5)
input.BackgroundColor3 = Color3.fromRGB(25,25,25)
input.TextColor3 = Color3.fromRGB(255,255,255)
input.PlaceholderText = "Key..."
input.Font = Enum.Font.Gotham
input.TextSize = 16
input.ClearTextOnFocus = false

Instance.new("UICorner", input).CornerRadius = UDim.new(0,10)

-- ENTER TO VERIFY
input.FocusLost:Connect(function(enterPressed)
	if not enterPressed then return end

	if input.Text ~= KEY then
		status.Text = "Invalid key"
		status.TextColor3 = Color3.fromRGB(255,90,90)
		task.wait(1.2)
		gui:Destroy()
		return
	end

	-- ĐÚNG KEY
	input:Destroy()
	status.TextColor3 = Color3.fromRGB(220,220,220)

	local function step(msg, t)
		status.Text = msg
		task.wait(t)
	end

	step("Verifying key...", 0.6)
	step("Checking environment...", 0.6)
	step("Loading modules...", 0.6)
	step("Finalizing...", 0.6)

	-- FADE OUT
	TweenService:Create(bg, TweenInfo.new(0.3), {BackgroundTransparency = 1}):Play()
	TweenService:Create(status, TweenInfo.new(0.3), {TextTransparency = 1}):Play()
	TweenService:Create(icon, TweenInfo.new(0.3), {ImageTransparency = 1}):Play()

	task.wait(0.35)
	gui:Destroy()

	-- LOAD MAIN SCRIPT
	loadstring(game:HttpGet(SCRIPT_URL))()
end)
